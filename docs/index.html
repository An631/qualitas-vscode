Easy Configuration

Add a qualitas.config.js to your project root and the whole team shares the same rules

// qualitas.config.js
module.exports = {
  profile: "strict",
  threshold: 80,
  flags: {
    tooManyParams: { warn: 3, error: 5 },
    excessiveReturns: false,
  },
  languages: {
    typescript: {
      flags: { deepNesting: { warn: 3, error: 5 } }
    }
  }
};
